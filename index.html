const SPREADSHEET_ID = 'ใส่_ID_ชีตของคุณที่นี่'; const SHEET_NAME = 'Data'; function doPost(e) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); const data = JSON.parse(e.postData.contents); // โหมดบันทึกข้อมูล if (data.action === 'save') { sheet.appendRow([ new Date(), data.hn, data.patientName, data.bed, data.admDate, data.duration, data.motorPower, data.adlScore, data.thaiMedStatus ]); return ContentService.createTextOutput(JSON.stringify({status: 'success'})) .setMimeType(ContentService.MimeType.JSON); } // โหมดค้นหาข้อมูลคนไข้ (ดึงมาสร้างการ์ด) if (data.action === 'search') { const dataRange = sheet.getDataRange().getValues(); const headers = dataRange[0]; const rows = dataRange.slice(1); let results = []; rows.forEach(row => { // ค้นหาจาก HN หรือ ชื่อ (คอลัมน์ 1 คือ HN, 2 คือ ชื่อ) if (row[1].toString().includes(data.keyword) || row[2].toString().includes(data.keyword)) { results.push({ hn: row[1], name: row[2], bed: row[3], admDate: row[4], lastUpdate: row[0] }); } }); // กรองเอาเฉพาะข้อมูลล่าสุดของแต่ละ HN let uniqueResults = results.reverse().filter((v, i, a) => a.findIndex(t => (t.hn === v.hn)) === i); return ContentService.createTextOutput(JSON.stringify({status: 'success', data: uniqueResults})) .setMimeType(ContentService.MimeType.JSON); } }
